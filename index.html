-- Remove invalid location entries (name fragments) from Edmonton locations. -- Keeps only tokens shaped like real cells: LETTERS + number - number (e.g. E6-2, SHELF1-1) UPDATE products SET locations_edmonton = COALESCE( (SELECT jsonb_agg(elem) FROM jsonb_array_elements_text(locations_edmonton) AS elem WHERE elem ~ '^[A-Z]+[0-9]+-[0-9]+$'), '[]'::jsonb ) WHERE locations_edmonton IS NOT NULL AND locations_edmonton != '[]'::jsonb; -- Rebuild the raw location text from the cleaned array UPDATE products SET raw_location_edmonton = COALESCE( (SELECT string_agg(elem, ', ') FROM jsonb_array_elements_text(locations_edmonton) AS elem), '' ); -- Verify: this should return 0 rows SELECT sku, locations_edmonton FROM products WHERE EXISTS ( SELECT 1 FROM jsonb_array_elements_text(locations_edmonton) AS elem WHERE elem !~ '^[A-Z]+[0-9]+-[0-9]+$' );
